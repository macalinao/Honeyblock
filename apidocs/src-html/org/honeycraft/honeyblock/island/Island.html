086    /**
087     * Cleans up the entire area of the island.
088     */
089    public void cleanup() {
090        for (int i = getX() - ISLAND_SPACING_HALF; i < getX() + ISLAND_SPACING_HALF; i++) {
091            for (int j = 0; j < getWorld().getMaxHeight(); j++) {
092                for (int k = getZ() - ISLAND_SPACING_HALF; k < getZ() + ISLAND_SPACING_HALF; k++) {
093                    Block block = getWorld().getBlockAt(i, j, k);
094                    if (block.getTypeId() != 0) {
095                        block.setTypeId(0);
096                    }
097                }
098            }
099        }
100    }
